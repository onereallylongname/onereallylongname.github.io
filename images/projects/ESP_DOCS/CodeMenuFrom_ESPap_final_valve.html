void setup()
{
/*Setup IO*/
setUpIO();
digitalWrite(ENABLE, HIGH);

/*Chip Setup end*/
sayHello0();

/*Setup Serial for Debug*/
setUpSerialForDebug();

/*File system*/
setUpFileSystem();

/*Fun��o para o debug dos ficheiros*/
if (DEBUG) checkDir("/");

/*Iniciar ssid e password com informa��o do ficheiro de configura��o*/
startAPInfoFromConfig();

/*Configurar o ip do servidor com recurso ao ficheiro de configuração*/
startServerIPFromConfig();

/*Configurar o nome do terreno com recurso ao ficheiro de configuração*/
startServerFieldFromConfig();

/*Actualizar quais os erros activos com recurso ao ficheiro de configuração*/
startServerErrorFromConfig();

/*Start WiFi*/
setUpAndStartWiFi();

/*Setup server*/
setUpServerOn();

server.begin();

PrintDegug("\n>>HTTP server started...\n", 1);

/*Chip Setup end*/
sayHello0();
mainLoopT = millis();
}


Loop

dscr:Main loop in arduino
***************************************************************************/
void loop()
{
server.handleClient();
// Set warning Ligth
setLedDispInfo();
displayLed();
/* Modo manual */
if (digitalRead(SWITCH))
{
if (!cooldown)
{
if (digitalRead(VONOFF))
{
//PrintDegug("<- Button Down\n", 3);
if (!pressed)
{
pressed = true;
}
}
else if (pressed && !digitalRead(VONOFF))
{
irrigate(!irrigationON);
pressed = false;
cooldown = 25000;
PrintDegug(String(millis()) + "\n-- -- -- -- -- -- -- --\n ", 0);
}
}
else
if (cooldown > 0)
cooldown--;
}
/* Modo automatico */
else
{
if(millis() > mainLoopTimeOut && connectToWeb()){
readWaterFlow();
autoRunRoutine();
}
}
}
